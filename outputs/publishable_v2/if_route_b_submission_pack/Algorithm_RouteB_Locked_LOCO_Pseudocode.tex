\begin{algorithm}[htbp]
\small
\caption{Locked Route-B LOCO Benchmark and Score-Level Adaptation Analysis}
\label{alg:route_b_locked_loco}
\begin{algorithmic}[1]
    \State \textbf{Input:} Multicentre cohort $\mathcal{D}=\{(X^{col}_i,X^{oct}_i,r_i,s_i,c_i,y^{2+}_i,y^{3+}_i)\}_{i=1}^{N}$, centre set $\mathcal{C}$, frozen feature and prediction caches $\mathcal{F}$, threshold candidates $\mathcal{T}$, score-level TTA candidates $\mathcal{A}$, claim-lock checklist $\mathcal{Q}$.
    \State \textbf{Endpoints:} Primary endpoint $y^{2+}$ for CIN2+ ranking; safety endpoint $y^{3+}$ for CIN3+ sensitivity and false-negative analysis.
    \State \textbf{Modules:} Clinical baselines $B_{clin}$, unimodal and simple-fusion baselines $B_{base}$, source-only DG ensemble $E_{DG}$, score-level adaptation operators $\mathcal{A}$, domain-shift analyser $S_{dom}$, evaluation engine $M$, audit writer $W$.

    \Statex \Comment{\textbf{Phase 1: Protocol audit and split construction}}
    \State Verify $N=1897$, five-centre identity, CIN2+/CIN3+ labels, threshold columns, and required prediction/feature caches.
    \For{\textbf{each} held-out centre $c^{\star}\in\mathcal{C}$}
        \State $\mathcal{D}_{src}\gets\{i:c_i\neq c^{\star}\}$, \quad $\mathcal{D}_{tgt}\gets\{i:c_i=c^{\star}\}$.
        \State Lock protocol flags: source-only LOCO uses no target-centre labels or unlabelled target adaptation; transductive TTA may use unlabelled target scores/features but never target labels during adaptation.
    \EndFor

    \Statex \Comment{\textbf{Phase 2: Source-only LOCO benchmark}}
    \For{\textbf{each} held-out centre $c^{\star}\in\mathcal{C}$}
        \State Fit or load $B_{clin}$, $B_{base}$, and multimodal HyDRA-DG candidates using $\mathcal{D}_{src}$ only.
        \State Select the source-only DG ensemble $E_{DG}^{c^{\star}}$ by inner-source validation and the locked safety-aware rule.
        \State Predict source-only probabilities $p^{src}_i=E_{DG}^{c^{\star}}(\mathcal{F}_i)$ for all $i\in\mathcal{D}_{tgt}$.
        \State Store the source operating point $t^{src}_{c^{\star}}\in\mathcal{T}$ from the source/inner-validation protocol.
    \EndFor
    \State Aggregate source-only predictions $\mathbf{p}^{src}$ across the five held-out folds.
    \State Compute pooled and centre-level $\mathrm{AUC}_{2+}$, $\mathrm{AUC}_{3+}$, CIN3+ sensitivity, false negatives, screen-positive rate, and centre gap.

    \Statex \Comment{\textbf{Phase 3: Score-level transductive adaptation boundary}}
    \For{\textbf{each} held-out centre $c^{\star}\in\mathcal{C}$}
        \For{\textbf{each} score-level candidate $a\in\mathcal{A}$}
            \State Fit adaptation transform $g_a^{c^{\star}}$ using source scores/features and unlabelled target-centre score distribution only.
            \State Produce adapted probabilities $p^{a}_i=g_a^{c^{\star}}(p^{src}_i,\mathcal{F}_i)$ for all $i\in\mathcal{D}_{tgt}$.
            \State Store $p^{a}_i$ and the adapted operating point $t^{a}_{c^{\star}}$ without using target labels.
        \EndFor
    \EndFor
    \State After all predictions are locked, evaluate each $a\in\mathcal{A}$ with target labels for reporting only.
    \State Mark AdaptiveBN and TENT as \textsc{Not Available} if fold-specific image-level checkpoints are absent.
    \State Define $a^{\dagger}$ as the descriptive best score-level TTA candidate for boundary analysis, not as an inductive deployment model.

    \Statex \Comment{\textbf{Phase 4: Domain-shift and hard-centre diagnosis}}
    \State Build frozen patient-level multimodal feature matrix $Z$ from $\mathcal{F}$.
    \State $S_{dom}\gets\{\mathrm{MMD}(Z,c),\mathrm{UMAP}(Z),\mathrm{CentreClassifier}(Z),\mathrm{SourceAttribution}(Z)\}$.
    \State Identify hardest centre $c_{hard}$ by source-only centre-level safety/ranking failure, not by maximum MMD alone.
    \State For $c_{hard}$, compare $\mathbf{p}^{src}$ and $\mathbf{p}^{a^{\dagger}}$ by ROC, threshold sweep, calibration transform, CIN3+ false negatives, and case-level error summaries.

    \Statex \Comment{\textbf{Phase 5: Clinical utility, calibration, and claim lock}}
    \State Compute calibration curves, ECE, decision-curve analysis, and screening-efficiency curves for source-only and adapted scores.
    \State Generate tables and figures for protocol separation, dataset demographics, fusion ladder, centre-level results, TTA comparison, hard-centre analysis, MMD/UMAP, ROC/TTA, ranking-vs-calibration, and clinical evaluation.
    \State Apply claim-lock checklist $\mathcal{Q}$: allow benchmark and limitation claims; caveat selected-baseline and transductive-TTA claims; prohibit deployment-ready, image-level-TTA, and clinically validated CoE claims.
    \State Treat Chain-of-Evidence outputs as template-supervised transparency aids pending expert validation.

    \Statex \Comment{\textbf{Output}}
    \State \textbf{Output:} Locked source-only Route-B benchmark, score-level TTA boundary results, hard-centre failure diagnosis, calibrated clinical evaluation, manuscript tables/figures, and final claim-lock audit.
\end{algorithmic}
\end{algorithm}
